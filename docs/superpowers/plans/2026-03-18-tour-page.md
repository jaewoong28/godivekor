# Tour Page Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a bilingual (KO/EN) diving tour page with a left scrollable point card list and a right sticky Leaflet interactive map that smoothly pans to clicked points.

**Architecture:** New `/[locale]/tours` page using react-leaflet for the interactive map. Dive point data stored in `data/divePoints.ts`. Left panel shows country-grouped point cards, right panel is a sticky Leaflet map. Clicking a card triggers `map.flyTo()` for smooth animation. Mobile: map on top, cards below. Header nav updated with "투어" link.

**Tech Stack:** Next.js 16, React 19, react-leaflet, leaflet, Tailwind CSS 4, next-intl

**Spec:** Brainstorming session 2026-03-18 — approved mockup: `tour-page-v2.html`

export function getCountryGroups(): CountryGroup[] {
  const grouped = new Map<string, DivePoint[]>();
  for (const point of divePoints) {
    const existing = grouped.get(point.country) || [];
    existing.push(point);
    grouped.set(point.country, existing);
  }
  return countryOrder
    .filter((c) => grouped.has(c))
    .map((c) => {
      const points = grouped.get(c)!;
      return {
        country: c,
        countryCode: points[0].countryCode,
        flag: points[0].flag,
        points,
      };
    });
}

"use client";

import { useTranslations } from "next-intl";
import type { DivePoint } from "@/data/divePoints";

interface TourPointCardProps {
  point: DivePoint;
  isActive: boolean;
  onClick: () => void;
}

export default function TourPointCard({ point, isActive, onClick }: TourPointCardProps) {
  const t = useTranslations("Tours.points");
  const tags = t(`${point.id}.tags`).split(",");

  return (

        />
        <div className="flex-1 min-w-0">
          <h3 className="text-sm font-bold text-[#111111] truncate">

```tsx
"use client";

import { useTranslations } from "next-intl";
import type { CountryGroup, DivePoint } from "@/data/divePoints";
import TourPointCard from "./TourPointCard";

interface TourCardListProps {
  countryGroups: CountryGroup[];
  activePointId: string | null;
  onPointClick: (point: DivePoint) => void;
}

export default function TourCardList({
  countryGroups,
  activePointId,
  onPointClick,
}: TourCardListProps) {
  const t = useTranslations("Tours");

  return (
    <div className="h-full overflow-y-auto px-4 sm:px-6 py-6 space-y-6">
      {countryGroups.map((group) => (
        <div key={group.country} id={`country-${group.country}`}>
          {/* Country header */}
          <div className="flex items-center gap-3 pb-3 mb-3 border-b-2 border-sunset/30">
            <span className="text-2xl">{group.flag}</span>
            <h2 className="text-lg font-extrabold text-[#111111]">
              {t(`countries.${group.country}`)}
            </h2>
            <span className="text-xs text-sunset font-semibold ml-auto">
              {t("pointCount", { count: group.points.length })}
            </span>
          </div>

          {/* Point cards */}
          <div className="space-y-3">
            {group.points.map((point) => (
              <TourPointCard
                key={point.id}
                point={point}
                isActive={activePointId === point.id}
                onClick={() => onPointClick(point)}
              />
            ))}
          </div>
        </div>
      ))}
    </div>
  );
}

  CircleMarker,
  Popup,
  useMap,
} from "react-leaflet";
import { useTranslations } from "next-intl";
import type { DivePoint } from "@/data/divePoints";

// Subcomponent to handle flyTo animation
function FlyToPoint({ point }: { point: DivePoint | null }) {
  const map = useMap();
  useEffect(() => {
    if (point) {
      map.flyTo([point.lat, point.lng], 8, { duration: 1.5 });
    }
  }, [point, map]);
  return null;
}

interface TourMapProps {
  points: DivePoint[];
  activePoint: DivePoint | null;
  onMarkerClick: (point: DivePoint) => void;
}

export default function TourMap({ points, activePoint, onMarkerClick }: TourMapProps) {
  const t = useTranslations("Tours.points");

  return (
    <MapContainer
      center={[15, 125]}
      zoom={4}
      className="w-full h-full"
      zoomControl={false}
      attributionControl={false}
    >
      <TileLayer
        url="https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png"
        attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'
      />
      <FlyToPoint point={activePoint} />
      {points.map((point) => {
        const isActive = activePoint?.id === point.id;
        return (
          <CircleMarker
            key={point.id}
            center={[point.lat, point.lng]}
            radius={isActive ? 10 : 6}
            pathOptions={{
              color: point.markerColor,
              fillColor: point.markerColor,
              fillOpacity: isActive ? 0.9 : 0.6,
              weight: isActive ? 3 : 1,
            }}
            eventHandlers={{
              click: () => onMarkerClick(point),
            }}
          >
            <Popup>
              <div className="text-xs">
                <p className="font-bold">{t(`${point.id}.name`)}</p>
                <p className="text-gray-500 mt-1">{t(`${point.id}.description`)}</p>
              </div>
            </Popup>
          </CircleMarker>
        );
      })}
    </MapContainer>
  );
}
```

"use client";

import { useState, useMemo } from "react";
import dynamic from "next/dynamic";
import { useTranslations } from "next-intl";
import { divePoints, getCountryGroups } from "@/data/divePoints";
import type { DivePoint } from "@/data/divePoints";
import TourCardList from "./TourCardList";

// Dynamic import to avoid SSR with Leaflet
const TourMap = dynamic(() => import("./TourMap"), {
  ssr: false,
  loading: () => (
    <div className="w-full h-full bg-[#0d2137] flex items-center justify-center">
      <div className="text-ocean-light text-sm">지도 로딩 중...</div>
    </div>
  ),
});

export default function TourPage() {
  const [activePoint, setActivePoint] = useState<DivePoint | null>(null);
  const t = useTranslations("Tours");
  const countryGroups = useMemo(() => getCountryGroups(), []);

  const handlePointClick = (point: DivePoint) => {
    setActivePoint(point);
  };

  const handleMarkerClick = (point: DivePoint) => {
    setActivePoint(point);
    // Scroll to the card in the left panel
    const el = document.getElementById(`country-${point.country}`);
    el?.scrollIntoView({ behavior: "smooth", block: "start" });
  };

  return (
    <main className="pt-16">
      {/* Page header */}
      <div className="text-center py-8 bg-white border-b border-gray-100">
        <h1 className="text-2xl sm:text-3xl font-black text-[#111111]">
          {t("pageTitle")}
        </h1>
        <p className="text-sm text-gray-500 mt-2">{t("pageSubtitle")}</p>
      </div>

      {/* Desktop: side-by-side | Mobile: stacked */}
      <div className="flex flex-col lg:flex-row" style={{ height: "calc(100vh - 170px)" }}>
        {/* Left: Card list */}
        <div className="lg:w-1/2 overflow-y-auto bg-white order-2 lg:order-1">
          <TourCardList
            countryGroups={countryGroups}
            activePointId={activePoint?.id ?? null}
            onPointClick={handlePointClick}
          />
        </div>

        {/* Right: Sticky map */}
        <div className="lg:w-1/2 h-[40vh] lg:h-full sticky top-16 order-1 lg:order-2">
          <TourMap
            points={divePoints}
            activePoint={activePoint}
            onMarkerClick={handleMarkerClick}
          />
        </div>
      </div>
    </main>
  );
}
